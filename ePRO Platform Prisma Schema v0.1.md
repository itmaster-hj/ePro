```prisma
///////////////////////////////////////////////////////////////
// ePRO Platform Prisma Schema v0.1
//
// 想定環境:
// - PostgreSQL
// - Prisma v6
// - Next.js App Router
// - Docker
// - AWS EC2
// - Formbricks
//
// 設計方針:
// - medical_auth
// - medical_private
// - medical_research
//
// の3schema分離
//
// PII（個人情報）は medical_private に隔離
//
// 監査ログ対応
// 論理削除対応
// 将来的なEMR連携考慮
//
///////////////////////////////////////////////////////////////

///////////////////////////////////////////////////////////////
// Generator
///////////////////////////////////////////////////////////////

generator client {
  provider        = "prisma-client-js"

  // PostgreSQL Multi Schema 対応
  previewFeatures = ["multiSchema"]
}

///////////////////////////////////////////////////////////////
// Datasource
///////////////////////////////////////////////////////////////

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")

  // PostgreSQL schema
  schemas = [
    "medical_auth",
    "medical_private",
    "medical_research"
  ]
}

///////////////////////////////////////////////////////////////
// ENUMS
///////////////////////////////////////////////////////////////

///////////////////////////////////////////////////////////////
// 組織種別
///////////////////////////////////////////////////////////////

enum OrganizationType {
  hospital
  university
  pharma

  @@schema("medical_auth")
}

///////////////////////////////////////////////////////////////
// 研究ステータス
///////////////////////////////////////////////////////////////

enum ProjectStatus {
  draft
  active
  closed

  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// Survey配信方法
///////////////////////////////////////////////////////////////

enum DeliveryType {
  email
  sms
  line

  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// Survey送信状態
///////////////////////////////////////////////////////////////

enum SurveyDeliveryStatus {
  scheduled
  sent
  opened
  completed
  failed

  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// 通知状態
///////////////////////////////////////////////////////////////

enum NotificationStatus {
  queued
  sent
  opened
  failed

  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// 謝礼状態
///////////////////////////////////////////////////////////////

enum RewardStatus {
  pending
  granted
  failed

  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// medical_auth schema
///////////////////////////////////////////////////////////////

///////////////////////////////////////////////////////////////
// 組織
//
// 病院
// 大学
// 製薬企業
///////////////////////////////////////////////////////////////

model Organization {
  /// 組織ID
  id String @id @default(uuid()) @db.Uuid

  /// 組織種別
  organizationType OrganizationType

  /// 組織名
  name String

  /// 有効フラグ
  active Boolean @default(true)

  /// 作成日時
  createdAt DateTime @default(now())

  /// 更新日時
  updatedAt DateTime @updatedAt

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  /// 所属ユーザー
  users User[]

  /// 研究プロジェクト
  projects ResearchProject[]

  /////////////////////////////////////////////////////////////
  // DB Mapping
  /////////////////////////////////////////////////////////////

  @@map("organizations")
  @@schema("medical_auth")
}

///////////////////////////////////////////////////////////////
// 管理ユーザー
//
// CRC
// 医師
// 管理者
///////////////////////////////////////////////////////////////

model User {
  /// ユーザーID
  id String @id @default(uuid()) @db.Uuid

  /// 所属組織
  organizationId String @db.Uuid

  /// ログインメール
  email String @unique

  /// bcrypt hash
  passwordHash String

  /// MFA有効
  mfaEnabled Boolean @default(false)

  /// 有効フラグ
  active Boolean @default(true)

  /// 論理削除
  deletedAt DateTime?

  /// 作成日時
  createdAt DateTime @default(now())

  /// 更新日時
  updatedAt DateTime @updatedAt

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  organization Organization
    @relation(fields: [organizationId], references: [id])

  /// ユーザーロール
  userRoles UserRole[]

  /// 監査ログ
  auditLogs AuditLog[]

  /// 作成した研究
  createdProjects ResearchProject[]

  /////////////////////////////////////////////////////////////
  // Index
  /////////////////////////////////////////////////////////////

  @@index([organizationId])

  /////////////////////////////////////////////////////////////
  // DB Mapping
  /////////////////////////////////////////////////////////////

  @@map("users")
  @@schema("medical_auth")
}

///////////////////////////////////////////////////////////////
// ロール
///////////////////////////////////////////////////////////////

model Role {
  /// ロールID
  id String @id @default(uuid()) @db.Uuid

  /// ロール名
  /// super_admin
  /// study_admin
  /// crc
  /// investigator
  roleName String @unique

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  userRoles UserRole[]

  @@map("roles")
  @@schema("medical_auth")
}

///////////////////////////////////////////////////////////////
// ユーザーとロールの中間テーブル
///////////////////////////////////////////////////////////////

model UserRole {
  id String @id @default(uuid()) @db.Uuid

  userId String @db.Uuid

  roleId String @db.Uuid

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  user User
    @relation(fields: [userId], references: [id])

  role Role
    @relation(fields: [roleId], references: [id])

  /////////////////////////////////////////////////////////////
  // Constraints
  /////////////////////////////////////////////////////////////

  @@unique([userId, roleId])

  @@index([userId])
  @@index([roleId])

  @@map("user_roles")
  @@schema("medical_auth")
}

///////////////////////////////////////////////////////////////
// 監査ログ
//
// 重要操作を記録
///////////////////////////////////////////////////////////////

model AuditLog {
  id String @id @default(uuid()) @db.Uuid

  /// 実行ユーザー
  userId String? @db.Uuid

  /// 操作種別
  /// PATIENT_CREATE 等
  action String

  /// 対象テーブル
  targetType String?

  /// 対象ID
  targetId String? @db.Uuid

  /// IPアドレス
  ipAddress String?

  /// UserAgent
  userAgent String?

  /// 実行日時
  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  user User?
    @relation(fields: [userId], references: [id])

  /////////////////////////////////////////////////////////////
  // Index
  /////////////////////////////////////////////////////////////

  @@index([userId])
  @@index([action])

  @@map("audit_logs")
  @@schema("medical_auth")
}

///////////////////////////////////////////////////////////////
// medical_private schema
//
// PII領域
///////////////////////////////////////////////////////////////

///////////////////////////////////////////////////////////////
// 患者
///////////////////////////////////////////////////////////////

model Patient {
  /// 内部ID
  id String @id @default(uuid()) @db.Uuid

  /// システム横断UUID
  patientUuid String @unique @db.Uuid

  /// 医療機関ID
  facilityId String @db.Uuid

  /// 電子カルテ患者ID
  emrPatientId String?

  /////////////////////////////////////////////////////////////
  // PII
  // 暗号化推奨
  /////////////////////////////////////////////////////////////

  fullName String?

  birthDate DateTime?

  gender String?

  /////////////////////////////////////////////////////////////
  // 論理削除
  /////////////////////////////////////////////////////////////

  deletedAt DateTime?

  /////////////////////////////////////////////////////////////
  // Timestamp
  /////////////////////////////////////////////////////////////

  createdAt DateTime @default(now())

  updatedAt DateTime @updatedAt

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  contacts PatientContact[]

  magicLinks PatientMagicLink[]

  consents PatientConsent[]

  projectPatients ProjectPatient[]

  surveyDeliveries SurveyDelivery[]

  notifications Notification[]

  rewardLogs RewardLog[]

  /////////////////////////////////////////////////////////////
  // Index
  /////////////////////////////////////////////////////////////

  @@index([patientUuid])

  @@index([facilityId])

  @@map("patients")
  @@schema("medical_private")
}

///////////////////////////////////////////////////////////////
// 患者連絡先
///////////////////////////////////////////////////////////////

model PatientContact {
  id String @id @default(uuid()) @db.Uuid

  patientUuid String @db.Uuid

  /////////////////////////////////////////////////////////////
  // 暗号化推奨
  /////////////////////////////////////////////////////////////

  email String?

  phoneNumber String?

  /////////////////////////////////////////////////////////////
  // メール認証日時
  /////////////////////////////////////////////////////////////

  emailVerifiedAt DateTime?

  createdAt DateTime @default(now())

  updatedAt DateTime @updatedAt

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  patient Patient
    @relation(fields: [patientUuid], references: [patientUuid])

  @@index([patientUuid])

  @@map("patient_contacts")
  @@schema("medical_private")
}

///////////////////////////////////////////////////////////////
// マジックリンク
///////////////////////////////////////////////////////////////

model PatientMagicLink {
  id String @id @default(uuid()) @db.Uuid

  patientUuid String @db.Uuid

  /// ハッシュ化トークン
  tokenHash String

  /// 有効期限
  expiresAt DateTime

  /// 使用日時
  usedAt DateTime?

  /// IP
  ipAddress String?

  /// UA
  userAgent String?

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  patient Patient
    @relation(fields: [patientUuid], references: [patientUuid])

  @@index([patientUuid])

  @@index([expiresAt])

  @@map("patient_magic_links")
  @@schema("medical_private")
}

///////////////////////////////////////////////////////////////
// 電子同意
///////////////////////////////////////////////////////////////

model PatientConsent {
  id String @id @default(uuid()) @db.Uuid

  patientUuid String @db.Uuid

  /// 同意書Version
  consentVersion String

  /// 同意日時
  agreedAt DateTime?

  /// 撤回日時
  revokedAt DateTime?

  /// 同意取得時IP
  ipAddress String?

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  patient Patient
    @relation(fields: [patientUuid], references: [patientUuid])

  @@index([patientUuid])

  @@map("patient_consents")
  @@schema("medical_private")
}

///////////////////////////////////////////////////////////////
// medical_research schema
///////////////////////////////////////////////////////////////

///////////////////////////////////////////////////////////////
// 研究プロジェクト
///////////////////////////////////////////////////////////////

model ResearchProject {
  id String @id @default(uuid()) @db.Uuid

  organizationId String @db.Uuid

  /// プロジェクト名
  projectName String

  /// 説明
  description String?

  /// 状態
  status ProjectStatus

  /// 作成ユーザー
  createdBy String? @db.Uuid

  /// 論理削除
  deletedAt DateTime?

  createdAt DateTime @default(now())

  updatedAt DateTime @updatedAt

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  organization Organization
    @relation(fields: [organizationId], references: [id])

  creator User?
    @relation(fields: [createdBy], references: [id])

  projectPatients ProjectPatient[]

  surveys Survey[]

  /////////////////////////////////////////////////////////////
  // Index
  /////////////////////////////////////////////////////////////

  @@index([organizationId])

  @@index([status])

  @@map("research_projects")
  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// 研究参加患者
///////////////////////////////////////////////////////////////

model ProjectPatient {
  id String @id @default(uuid()) @db.Uuid

  projectId String @db.Uuid

  patientUuid String @db.Uuid

  /// 参加日時
  enrolledAt DateTime?

  /// 中止日時
  withdrawnAt DateTime?

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  project ResearchProject
    @relation(fields: [projectId], references: [id])

  patient Patient
    @relation(fields: [patientUuid], references: [patientUuid])

  /////////////////////////////////////////////////////////////
  // Constraints
  /////////////////////////////////////////////////////////////

  @@unique([projectId, patientUuid])

  @@index([projectId])

  @@index([patientUuid])

  @@map("project_patients")
  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// Survey
///////////////////////////////////////////////////////////////

model Survey {
  id String @id @default(uuid()) @db.Uuid

  projectId String @db.Uuid

  /// Formbricks Survey ID
  formbricksSurveyId String

  surveyName String

  active Boolean @default(true)

  deletedAt DateTime?

  createdAt DateTime @default(now())

  updatedAt DateTime @updatedAt

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  project ResearchProject
    @relation(fields: [projectId], references: [id])

  deliveries SurveyDelivery[]

  /////////////////////////////////////////////////////////////
  // Index
  /////////////////////////////////////////////////////////////

  @@index([projectId])

  @@map("surveys")
  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// Survey配信
///////////////////////////////////////////////////////////////

model SurveyDelivery {
  id String @id @default(uuid()) @db.Uuid

  patientUuid String @db.Uuid

  surveyId String @db.Uuid

  /// 配信方式
  deliveryType DeliveryType

  sentAt DateTime?

  openedAt DateTime?

  completedAt DateTime?

  /// 配信状態
  status SurveyDeliveryStatus

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  patient Patient
    @relation(fields: [patientUuid], references: [patientUuid])

  survey Survey
    @relation(fields: [surveyId], references: [id])

  response SurveyResponse?

  rewards RewardLog[]

  /////////////////////////////////////////////////////////////
  // Index
  /////////////////////////////////////////////////////////////

  @@index([patientUuid])

  @@index([surveyId])

  @@index([status])

  @@map("survey_deliveries")
  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// Survey回答
///////////////////////////////////////////////////////////////

model SurveyResponse {
  id String @id @default(uuid()) @db.Uuid

  surveyDeliveryId String @unique @db.Uuid

  /// Formbricks Response ID
  formbricksResponseId String

  respondedAt DateTime?

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  surveyDelivery SurveyDelivery
    @relation(fields: [surveyDeliveryId], references: [id])

  @@map("survey_responses")
  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// 通知履歴
///////////////////////////////////////////////////////////////

model Notification {
  id String @id @default(uuid()) @db.Uuid

  patientUuid String @db.Uuid

  /// 通知種別
  type String

  /// 件名
  subject String

  sentAt DateTime?

  openedAt DateTime?

  status NotificationStatus

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  patient Patient
    @relation(fields: [patientUuid], references: [patientUuid])

  @@index([patientUuid])

  @@map("notifications")
  @@schema("medical_research")
}

///////////////////////////////////////////////////////////////
// 謝礼ログ
///////////////////////////////////////////////////////////////

model RewardLog {
  id String @id @default(uuid()) @db.Uuid

  patientUuid String @db.Uuid

  surveyDeliveryId String @db.Uuid

  /// 謝礼種別
  rewardType String

  /// 金額
  rewardAmount Int

  /// 送信日時
  sentAt DateTime?

  /// 状態
  status RewardStatus

  createdAt DateTime @default(now())

  /////////////////////////////////////////////////////////////
  // Relations
  /////////////////////////////////////////////////////////////

  patient Patient
    @relation(fields: [patientUuid], references: [patientUuid])

  surveyDelivery SurveyDelivery
    @relation(fields: [surveyDeliveryId], references: [id])

  /////////////////////////////////////////////////////////////
  // Index
  /////////////////////////////////////////////////////////////

  @@index([patientUuid])

  @@index([surveyDeliveryId])

  @@map("reward_logs")
  @@schema("medical_research")
}
```
